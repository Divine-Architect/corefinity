STEP 3 — Tiny JS to sync pills with the field
Add a very small script block at the bottom of index.html, just before </body>, or into your existing JS file if there is one.
This script only powers the active state; the design degrades fine without it.

<script>
  (function () {
    const field = document.querySelector('.cf-fttb-field');
    const pills = document.querySelectorAll('.cf-fttb-pill');
    if (!field || !pills.length) return;

    // Default active
    if (!field.dataset.fttbActive) {
      const initial = document.querySelector('.cf-fttb-pill.is-active');
      const variant = initial?.dataset.fttbVariant || 'home';
      field.dataset.fttbActive = variant;
    }

    pills.forEach((pill) => {
      const variant = pill.dataset.fttbVariant;
      if (!variant) return;

      const activate = () => {
        pills.forEach((p) => {
          p.classList.toggle('is-active', p === pill);
          p.setAttribute('aria-selected', p === pill ? 'true' : 'false');
        });
        field.dataset.fttbActive = variant;
      };

      pill.addEventListener('mouseenter', activate);
      pill.addEventListener('focus', activate);
    });
  })();
</script>
